COMANDO: CREATE TABLE
	create table {NOME_TABELA} (
	  {NOME_COLUNA} {TIPO_DADO} {OPCOES} COMMENT {'comentário'}
	);


TIPOS DE DADOS {TIPO_DADO}
	Inteiro (Integer)
	Decimal/Numérico (Decimal/Numeric)
	Caracter/Varchar (Character/Varchar)
	Data/Hora (Date/Time)
	Booleano (Boolean)
	Texto longo (Texto)


OPÇÕES {OPCOES}
	Restrição de Valor:
		NOT NULL
		UNIQUE
		DEFAULT
	Chave primária e estrangeiras
	Auto Incremento
